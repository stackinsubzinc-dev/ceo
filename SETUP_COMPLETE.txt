🐍 PYTHON USAGE
═══════════════════════════════════════════════════════════════════════

import asyncio
from ai_services.product_tiktok_integration import get_product_tiktok_integration

async def launch_product():
    integration = get_product_tiktok_integration()
    
    product = {
        "id": "prod_123",
        "title": "AI Copywriting Course",
        "description": "Master AI writing tools",
        "price": 97,
        "category": "course"
    }
    
    # Post single video
    result = await integration.post_product_to_tiktok(product)
    print(f"Posted! Video ID: {result['video_id']}")
    
    # Post 5-video series
    series = await integration.post_product_series_to_tiktok(product, series_count=5)
    print(f"Posted {series['videos_posted']} videos!")
    
    # Schedule posts
    scheduled = await integration.schedule_product_posts(
        product,
        schedule_dates=[
            "2026-04-10T10:00:00Z",
            "2026-04-11T14:00:00Z"
        ]
    )
    print(f"Scheduled {scheduled['posts_scheduled']} posts!")

asyncio.run(launch_product())

🔧 TROUBLESHOOTING
═══════════════════════════════════════════════════════════════════════

Problem: "TikTok credentials not configured"
Solution: Add TIKTOK_ACCESS_TOKEN to .env and restart server

Problem: "Video upload failed"
Solution: Use MP4 format, check file size, ensure readable

Problem: "Rate limit exceeded"
Solution: Wait 1 hour or use fewer uploads

Problem: Posts showing as mock
Solution: Add real TikTok API credentials to .env
